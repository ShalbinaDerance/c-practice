#include <stdio.h>

int main()
{
    int N;
    scanf("%d",&N);
    
    int hours;
    int totalhours = 0;
    int heavydays = 0;
    for(int i=0;i<N;i++){
            scanf("%d",&hours);
            totalhours += hours;
            if(hours>3) {
               heavydays++;
         }
    }
         int overtimecost = totalhours * 200;
    
        
        
    printf("Total Overtime Hours:%d\n", totalhours);
    printf("Overtime Cost:%d\n", overtimecost);
    printf("Heavy Overtime Days:%d\n", heavydays);



    

    return 0;
}
