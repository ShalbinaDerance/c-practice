#include <stdio.h>
int main() {
    int N,loss;
    int totalloss=0, highloss=0;
    scanf("%d",&N);
    for(int i=0;i<N;i++){
        scanf("%d",&loss);
        totalloss+=loss;
        if(loss>100){
            highloss++;
        }
    }
        printf("Total loss:%d\n",totalloss);
        printf("highloss days:%d",highloss);
  
    return 0;
}
