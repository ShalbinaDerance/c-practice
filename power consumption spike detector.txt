#include <stdio.h>

int main()
{
    int units, N, totalpower = 0, spikehours = 0;
    scanf("%d", &N);
    for(int i=0;i<N;i++) {
        scanf("%d", &units);
        totalpower += units;
        if(units>5) {
            spikehours++;
        }
    }
    
    printf("Total Units:%d\n", totalpower);
    printf("Spike Hours:%d\n", spikehours);

    return 0;
}
