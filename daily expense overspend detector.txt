#include <stdio.h>
int main() {
    int n,i;
    int expense,total=0;
    int overspendDays=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&expense);
        total+=expense;
        if(expense>1000){
            overspendDays++;
        }
    }
    
    printf("total expense:%d\n",total);
    printf("overspenDays:%d",overspendDays);

    return 0;
}
