#include <stdio.h>
int main()
{
    int capacity, n;
    scanf("%d", &capacity);
    scanf("%d", &n);
    int items, overflow = 0;
    for(int i=0;i<n;i++){
        scanf("%d",&items);
        capacity -= items;
        if(capacity < 0 && overflow == 0) {
        overflow++;
    }
}
    printf("Remaining Capacity: %d\n", capacity);
    printf("Overflow Days: %d", overflow);
    return 0;
}
