#include <stdio.h>
int main()
{
    int capacity, N, i = 1, load;
    int safe = 0, failure = 0;
    scanf("%d%d", &capacity, &N);
    while(i <= N) {
        scanf("%d", &load);
        if(load <= capacity) 
            safe++;
        else
            failure++;
        i++;
    
        
        
    }
        
   
    printf("Safe Hours : %d\n", safe);
   
    printf("Failure Count: %d", failure);
    
    return 0;
}
