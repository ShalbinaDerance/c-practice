#include <stdio.h>

int main()
{
    
    int n;
    scanf("%d", &n);
    int price, max = 0, count = 0;
    for(int i = 0; i < n; i++){
        scanf("%d", &price);
        if(i == 0 || price > max) {
            max = price;
        }
        if(price > 100) {
            count++;
        }
    }
    printf("Highest Price: %d\n", max);
    printf("High Price Days: %d\n", count);

    return 0;
}
