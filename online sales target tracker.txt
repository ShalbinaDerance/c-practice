#include <stdio.h>

int main()
{
    
    int n;
    scanf("%d", &n);
    int dailysales, totalrevenue = 0, countdays = 0;
    for(int i = 0; i < n; i++){
        scanf("%d", &dailysales);
        if(dailysales <= 50000) {
            totalrevenue += dailysales;
            countdays++;
        }
    }
    printf("Total Revenue: %d\n", totalrevenue);
    printf("target Days: %d\n", countdays);

    return 0;
}
