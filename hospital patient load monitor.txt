#include <stdio.h>
int main() {
    int N,patients;
    int totalpatients=0,overcrowdedDays=0;
    scanf("%d",&N);
    for(int i=0;i<N;i++){
        scanf("%d",&patients);
        totalpatients+=patients;
        if(patients>100){
            overcrowdedDays++;
            
        }
    }
        printf("total patients:%d\n",totalpatients);
        printf("overcrowdedDays:%d",overcrowdedDays);
  
    return 0;
}
