#include <stdio.h>

int main()
{
    int callcenter, N, totalcalls = 0, overloadhours = 0;
    scanf("%d", &N);
    for(int i=0;i<N;i++) {
        scanf("%d", &callcenter);
        totalcalls += callcenter;
        if(callcenter>40) {
            overloadhours++;
        }
    }
    
    printf("Total Calls:%d\n", totalcalls);
    printf("Overloaded Hours:%d\n", overloadhours);

    return 0;
}
