#include <stdio.h>
int main()
{
    int fuel, N;
    int i = 0;
    int status;
    int consumption;
    int trips = 0;
    scanf("%d", &fuel);
    scanf("%d", &N);
    
    while(i < N) {
        scanf("%d", &consumption);
        if(fuel >= consumption) {
            fuel -= consumption;
            trips++;
        } else {
            break;
        }
        i++;
        
    }
    printf("Completed Trips: %d\n", trips);
    printf("Remaining Fuel: %d\n", fuel);
   
 


    return 0;
}
