#include <stdio.h>

int main()
{
    int n;
    scanf("%d",&n);
    int delays,totaldelay = 0, countdeliver = 0;
    for(int i = 0; i < n; i++){
    scanf("%d",&delays);
        totaldelay+=delays;
        if(delays>=30)
        countdeliver++;
    }
        
        
    printf("Total Delay:%d\n", totaldelay);
    printf("Delayed Deliveries:%d\n", countdeliver);



    

    return 0;
}
